    Type Alias Arc56Contract

    Describes the entire contract. This type is an extension of the type described in ARC-4

    type Arc56Contract = {
        arcs: number[];
        bareActions: {
            call: (
                | "NoOp"
                | "OptIn"
                | "CloseOut"
                | "ClearState"
                | "UpdateApplication"
                | "DeleteApplication"
            )[];
            create: ("NoOp" | "OptIn" | "DeleteApplication")[];
        };
        byteCode?: { approval: string; clear: string };
        compilerInfo?: {
            compiler: "algod" | "puya";
            compilerVersion: {
                commitHash?: string;
                major: number;
                minor: number;
                patch: number;
            };
        };
        desc?: string;
        events?: Event[];
        methods: Arc56Method[];
        name: string;
        networks?: { [network: string]: { appID: number } };
        scratchVariables?: {
            [name: string]: {
                slot: number;
                type: ABITypeName | AVMType | StructName;
            };
        };
        source?: { approval: string; clear: string };
        sourceInfo?: { approval: ProgramSourceInfo; clear: ProgramSourceInfo };
        state: {
            keys: {
                box: { [name: string]: StorageKey };
                global: { [name: string]: StorageKey };
                local: { [name: string]: StorageKey };
            };
            maps: {
                box: { [name: string]: StorageMap };
                global: { [name: string]: StorageMap };
                local: { [name: string]: StorageMap };
            };
            schema: {
                global: { bytes: number; ints: number };
                local: { bytes: number; ints: number };
            };
        };
        structs: { [structName: string]: StructField[] };
        templateVariables?: {
            [name: string]: {
                type: ABITypeName | AVMType | StructName;
                value?: string;
            };
        };
    }

    networks?: { [network: string]: { appID: number } }

    Optional object listing the contract instances across different networks. The key is the base64 genesis hash of the network, and the value contains information about the deployed contract in the network indicated by the key. A key containing the human-readable name of the network MAY be included, but the corresponding genesis hash key MUST also be defined
